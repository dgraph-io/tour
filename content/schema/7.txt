upsert {
  query {
    companyabc(func: eq(name, "CompanyABC")) {
      companyabc as uid
    }
    michael(func: eq(name, "Michael")) {
      michael as uid
    }
    sarah(func: eq(name, "Sarah")) {
      sarah as uid
    }
  }

  mutation {
    set {
      uid(michael) <boss_of> uid(sarah) .
      uid(sarah) <works_for> uid(companyabc) .
    }
  }
}
